838 //
839 // A string tokenizer taken from http://www.sunsite.ualberta.ca/Documentation/
840 // /Gnu/libstdc++-2.90.8/html/21_strings/stringtok_std_h.txt
841 //
842 template <typename Container>
843 void stringtok (Container &container,
844  std::string const &in,
845  const char * const delimiters = " \t\n")
846 {
847  const std::string::size_type len = in.length();
848  std::string::size_type i = 0;
849 
850  while ( i < len )
851  {
852  // eat leading whitespace
853  i = in.find_first_not_of (delimiters, i);
854 
855  if (i == std::string::npos)
856  return; // nothing left but white space
857 
858  // find the end of the token
859  std::string::size_type j = in.find_first_of (delimiters, i);
860 
861  // push token
862  if (j == std::string::npos)
863  {
864  container.push_back (in.substr(i));
865  return;
866  }
867  else
868  container.push_back (in.substr(i, j-i));
869 
870  // set up for next loop
871  i = j + 1;
872  }
873 
874  return;
875 }
